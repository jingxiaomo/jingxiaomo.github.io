代码：

typedef long long ll;
#define INF 0x3f3f3f3f3f3f3f3f
ll ns[100010];
ll nx[100010][10];
ll ms[100010];
ll mx[100010][10];
ll nn[100010],mm[100010];
int main()
{
    int t,n,m,num;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&n,&m,&num);
        for(int i=0;i<n;i++)
        {
            scanf("%lld",&ns[i]);
            for(int j=0;j<num;j++)
                scanf("%lld",&nx[i][j]);
        }
        for(int i=0;i<m;i++)
        {
            scanf("%lld",&ms[i]);
            for(int j=0;j<num;j++)
                scanf("%lld",&mx[i][j]);
        }
        ll ans=-INF;
        for(int i=0;i<(1<<num);i++)
        {
            for(int j=0;j<n;j++)
                nn[j]=ns[j];
            for(int j=0;j<m;j++)
                mm[j]=-ms[j];
            for(int j=0;j<num;j++)
            {
                //printf("%d",i>>j&1);
                if((i>>j&1)==1)//+
                {
                    for(int k=0;k<n;k++)
                        nn[k]+=nx[k][j];
                    for(int k=0;k<m;k++)
                        mm[k]+=mx[k][j];
                }
                else
                {
                    for(int k=0;k<n;k++)
                        nn[k]-=nx[k][j];
                    for(int k=0;k<m;k++)
                        mm[k]-=mx[k][j];
                }
            }
            //printf("\n");
            ll MAX=-INF,MIN=INF;
            for(int j=0;j<n;j++)
                MAX=max(MAX,nn[j]);
            for(int j=0;j<m;j++)
                MIN=min(MIN,mm[j]);
            ans=max(ans,MAX-MIN);
        }
        printf("%lld\n",ans);
    }
    return 0;
}
