typedef long long ll;
struct mat
{
    long long a[2][2];
}m,r;
ll p;
mat mult(mat x,mat y)
{
    mat res={0};
    int i,j,k;
    for(i=0;i<2;i++)
        for(j=0;j<2;j++)
            for(k=0;k<2;k++)
                res.a[i][j]=(res.a[i][j]+(x.a[i][k]*y.a[k][j])%p+p)%p;
    return res;
}
mat PowerMod(mat x,long long n)
{
    mat ans={0};
    int i;
    for(i=0;i<2;i++)
        ans.a[i][i]=1;
    while(n>0)
    {
        if(n%2==1)
            ans=mult(ans,x);
        x=mult(x,x);
        n=n/2;
    }
    return ans;
}
int main()
{
    ll x,y,n;
    while(scanf("%lld%lld%lld%lld",&x,&y,&n,&p)!=EOF)

        m.a[0][0]=2*x;m.a[0][1]=y-x*x;

        ll ans=(r.a[0][0]*2*x+r.a[0][1]*2)%p;
        printf("%lld\n",ans);
    }
    return 0;

long long PowerMod(long long a,long long b,long long c)
{
    long long ans=1;
    long long k;
    k=a;
    k=k%c;
    while(b>0)
    {

        ans=(ans*k)%c;

        k=(k*k)%c;
    }
    return ans;
}
