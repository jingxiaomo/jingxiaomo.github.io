题意：

typedef long long ll;
ll f[100010];
int main()
{
    ll n,m;
    f[0]=1;f[1]=0;f[2]=1;f[3]=1;
    while(scanf("%lld%lld",&n,&m)!=EOF)
    {
        if(n<=3){printf("%lld\n",f[n]);continue;}
        for(ll i=4;i<=n;i++)
            f[i]=(((i-1)*(f[i-1]+f[i-2])%m)%m-(((i-1)*(i-2)/2)%m*f[i-3])%m+m)%m;
        printf("%lld\n",f[n]);
    }
    return 0;
}
